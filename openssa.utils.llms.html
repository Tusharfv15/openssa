This class provides a consistent API for the different LLM services. Intended usage:

from openssa.utils.llms import OpenAILLM, AitomaticLLM, AzureLLM

llm1 = OpenAILLM.get_default() llm1.call(messages=[{“role”: “user”, “content”: “Say this is a test”}], stream=True) llm1.create_embeddings()

llm2 = AitomaticLLM.get_default() llm3 = AitoLLM.get_gpt_35_turbo_0613()

llm4 = AzureLLM.get_default()
